1 /**
  2  * Autoloads a file into the Node REPL on start then passes control to the user.
  3  * @author <a href="mailto:matthewkastor@gmail.com">
  4  *  Matthew Christopher Kastor-Inare III </a><br />
  5  *  ☭ Hial Atropa!! ☭
  6  * @version 20130309
  7  * @param {String} fileToLoad The file to load into the REPL
  8  * @param {String} encoding Optional. The encoding of the file, valid values
  9  *  are the same as for fs.readFileSync. Defaults to utf8.
 10  * @param {Object} replStartArgs Optional. The options for the REPL. Corresponds
 11  *  to repl.start options.
 12  * @see <a href="http://nodejs.org/api/fs.html#fs_fs_readfilesync_filename_encoding">
 13  * http://nodejs.org/api/fs.html#fs_fs_readfilesync_filename_encoding</a>
 14  * @see <a href="http://nodejs.org/api/repl.html#repl_repl_start_options">
 15  * http://nodejs.org/api/repl.html#repl_repl_start_options</a>
 16  */
 17 function replAutoload(fileToLoad, encoding, replStartArgs) {
 18     var repl = require("repl");
 19     var fs = require('fs');
 20     var file;
 21 
 22     encoding = encoding || 'utf8';
 23     replStartArgs = replStartArgs || 'atropa-replAutoload>';
 24     
 25     if(fileToLoad) {
 26         try {
 27             file = fs.readFileSync(fileToLoad, encoding);
 28             global.eval(String(file));
 29             console.log('Loaded ' + fileToLoad);
 30         } catch (e) {
 31             console.log('could not load file ' + fileToLoad + ' as ' + encoding);
 32             process.exit(1);
 33         }
 34     } else {
 35         console.log('No autoload file specified.');
 36     }
 37     
 38     repl.start(replStartArgs);
 39 };
 40 
 41 
 42 
 43 module.exports = replAutoload;
